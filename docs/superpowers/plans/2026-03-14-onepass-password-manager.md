# OnePass Password Manager Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a cross-platform (Android + macOS) personal password manager with local-only encrypted storage and LAN sync.

**Architecture:** Flutter monorepo compiling to both Android and macOS apps. All data stored in SQLCipher-encrypted SQLite. Sync is LAN-only: macOS runs an HTTP server discovered via mDNS; Android initiates bidirectional incremental sync.

**Tech Stack:** Flutter/Dart, SQLCipher (`sqflite_sqlcipher`), `pointycastle` (PBKDF2/HMAC), `local_auth` (biometrics), `shelf`/`shelf_router` (macOS HTTP server), `multicast_dns`, `riverpod`, `go_router`, `http`, `uuid`

```dart
import 'dart:convert';
import 'dart:math';
import 'dart:typed_data';
import 'package:pointycastle/export.dart';

class CryptoService {
  static const int _iterations = 100000;
  static const int _keyLength = 32; // 256-bit
  static const int _saltLength = 16;
  static const String _syncTokenMessage = 'onepass-sync-token';

  /// Derives a 256-bit key from [password] and [salt] using PBKDF2-SHA256.
  static Uint8List deriveKey(String password, Uint8List salt) {
    final pbkdf2 = PBKDF2KeyDerivator(HMac(SHA256Digest(), 64));
    pbkdf2.init(Pbkdf2Parameters(salt, _iterations, _keyLength));
    final passwordBytes = Uint8List.fromList(utf8.encode(password));
    return pbkdf2.process(passwordBytes);
  }

  /// Generates 16 random bytes suitable for use as a PBKDF2 salt.
  static Uint8List generateSalt() {
    final rng = Random.secure();
    return Uint8List.fromList(List.generate(_saltLength, (_) => rng.nextInt(256)));
  }

  /// Computes the sync auth token as HMAC-SHA256(key, "onepass-sync-token").
  /// Returns raw 32 bytes; caller base64-encodes for HTTP header.
  static Uint8List computeSyncToken(Uint8List keyBytes) {
    final hmac = HMac(SHA256Digest(), 64);
    hmac.init(KeyParameter(keyBytes));
    final message = Uint8List.fromList(_syncTokenMessage.codeUnits);
    return hmac.process(message);
  }
}
```

```dart
class VaultService {
  Database? _db;

  bool get isOpen => _db != null;

  /// Opens the encrypted database with the given key bytes.
  /// Call this after CryptoService.deriveKey() returns.
  Future<void> open(Uint8List keyBytes) async {
    final dir = await getApplicationSupportDirectory();
    final dbPath = p.join(dir.path, 'onepass.db');
    // SQLCipher expects the key as a hex string prefixed with "x'"
    final hexKey = "x'${hex.encode(keyBytes)}'";
    _db = await openDatabase(
      dbPath,
      version: DatabaseSchema.version,
      password: hexKey,
      onCreate: _createSchema,
      onOpen: (_) async => _seedDefaultCategory(),
    );
  }

  /// Opens an unencrypted in-memory database for unit/integration testing.
  Future<void> openForTesting() async {
    _db = await openDatabase(
      inMemoryDatabasePath,
      version: DatabaseSchema.version,
      onCreate: _createSchema,
      onOpen: (_) async => _seedDefaultCategory(),
    );
  }

  Future<void> _createSchema(Database db, int version) async {
    await db.execute(DatabaseSchema.createCategories);
    await db.execute(DatabaseSchema.createEntries);
    await db.execute(DatabaseSchema.createEntriesUpdatedAtIndex);
    await db.execute(DatabaseSchema.createCategoriesUpdatedAtIndex);
  }

  Future<void> _seedDefaultCategory() async {
    final existing = await _db!.query(
      'categories',
      where: 'id = ?',
      whereArgs: [Category.defaultCategoryId],
    );
    if (existing.isEmpty) {
      await upsertCategory(Category.defaultCategory);
    }
  }

  Future<void> close() async {
    await _db?.close();
    _db = null;
  }

  // ─── Entry CRUD ──────────────────────────────────────────────────────────

  Future<void> upsertEntry(Entry entry) async {
    await _db!.insert(
      'entries',
      _entryToRow(entry),
      conflictAlgorithm: ConflictAlgorithm.replace,
    );
  }

  Future<void> softDeleteEntry(String id) async {
    await _db!.update(
      'entries',
      {'deleted_at': DateTime.now().toUtc().toIso8601String()},
      where: 'id = ?',
      whereArgs: [id],
    );
  }

  /// Returns all non-deleted entries, ordered by title.
  Future<List<Entry>> getEntries() async {

      'entries',
      where: 'deleted_at IS NULL',
      orderBy: 'title ASC',

    final rows = await _db!.query(

      whereArgs: [since.toIso8601String()],
```

```dart
import 'package:local_auth/local_auth.dart';
import 'package:onepass_app/crypto/crypto_service.dart';

/// Holds the in-memory key bytes after successful unlock.
/// Cleared on lock.
class AuthService {
  final LocalAuthentication _localAuth = LocalAuthentication();

  Uint8List? _keyBytes;

  bool get isUnlocked => _keyBytes != null;

  Uint8List get keyBytes {
    if (_keyBytes == null) throw StateError('Vault is locked');
    return _keyBytes!;
  }

  /// Derives key from [password] + [salt], stores it in memory.
  /// Returns the derived key bytes.
  Uint8List deriveAndStoreKey(String password, Uint8List salt) {
    _keyBytes = CryptoService.deriveKey(password, salt);
    return _keyBytes!;
  }

  /// Saves key bytes into the platform secure storage for biometric unlock.
  /// On Android: Android Keystore (AES-256-GCM wrapping).
  /// On macOS: Keychain (kSecAttrAccessibleWhenUnlockedThisDeviceOnly).
  ///
  /// Uses the platform channel defined in the native embedding.
  Future<void> storeKeyForBiometrics(Uint8List keyBytes) async {
    const channel = MethodChannel('com.onepass/keystore');
    await channel.invokeMethod<void>('storeKey', {
      'key': base64.encode(keyBytes),
    });
  }

  /// Loads key bytes from secure storage, protected by biometric prompt.
  /// Returns null if biometric auth fails or no key is stored.
  Future<Uint8List?> loadKeyWithBiometrics() async {
    final canCheck = await _localAuth.canCheckBiometrics;
    if (!canCheck) return null;

    final authenticated = await _localAuth.authenticate(
      localizedReason: 'Unlock OnePass',
      options: const AuthenticationOptions(biometricOnly: true),
    );
    if (!authenticated) return null;

    const channel = MethodChannel('com.onepass/keystore');
    final stored = await channel.invokeMethod<String>('loadKey');
    if (stored == null) return null;

    _keyBytes = base64.decode(stored);
    return _keyBytes;
  }

  /// Checks whether a biometric key is stored (i.e. user has set up biometrics).
  Future<bool> hasBiometricKey() async {
    const channel = MethodChannel('com.onepass/keystore');
    final result = await channel.invokeMethod<bool>('hasKey');
    return result ?? false;
  }

  /// Clears the in-memory key. Call on app background or explicit lock.
  void lock() {
    _keyBytes = null;
  }
}
```

```dart
import 'package:go_router/go_router.dart';
```

```dart
import 'package:onepass_app/providers/auth_provider.dart';
import 'package:onepass_app/providers/vault_provider.dart';

class UnlockScreen extends ConsumerStatefulWidget {
  const UnlockScreen({super.key});
  @override
  ConsumerState<UnlockScreen> createState() => _UnlockScreenState();
}

class _UnlockScreenState extends ConsumerState<UnlockScreen> {
  final _passwordController = TextEditingController();
  String? _error;
  bool _loading = false;
  bool _obscure = true;

  @override
  void dispose() {
    _passwordController.dispose();
    super.dispose();
  }

  Future<Uint8List> _getOrCreateSalt() async {
    final dir = await getApplicationSupportDirectory();
    final saltFile = File(p.join(dir.path, 'salt.bin'));
    if (await saltFile.exists()) {
      return saltFile.readAsBytes();
    }
    final salt = CryptoService.generateSalt();
    await saltFile.writeAsBytes(salt);
    return salt;
  }

  Future<void> _unlock() async {
    final password = _passwordController.text.trim();
    if (password.isEmpty) {
      setState(() => _error = 'Password is required');
      return;
    }
    setState(() { _loading = true; _error = null; });
    try {
      final salt = await _getOrCreateSalt();
      final auth = ref.read(authServiceProvider);
      final keyBytes = auth.deriveAndStoreKey(password, salt);

      final vault = ref.read(vaultServiceProvider);
      await vault.open(keyBytes);

      ref.read(isUnlockedProvider.notifier).state = true;
      if (mounted) context.go('/entries');
    } catch (e) {
      setState(() => _error = 'Incorrect password or vault error');
    } finally {
      if (mounted) setState(() => _loading = false);
    }
  }

  Future<void> _biometricUnlock() async {
    setState(() { _loading = true; _error = null; });
    try {
      final auth = ref.read(authServiceProvider);
      final keyBytes = await auth.loadKeyWithBiometrics();
      if (keyBytes == null) {
        setState(() => _error = 'Biometric authentication failed');
        return;
      }
      final vault = ref.read(vaultServiceProvider);
      await vault.open(keyBytes);
      ref.read(isUnlockedProvider.notifier).state = true;
      if (mounted) context.go('/entries');
    } catch (e) {
      setState(() => _error = 'Biometric unlock failed');
    } finally {
      if (mounted) setState(() => _loading = false);
    }
  }

  Widget build(BuildContext context) {
    return Scaffold(
      body: SafeArea(
        child: Padding(
          padding: const EdgeInsets.symmetric(horizontal: 32),
          child: Column(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              const Icon(Icons.lock_outline, size: 64, color: Color(0xFF0A84FF)),
              const SizedBox(height: 24),
              const Text('OnePass', style: TextStyle(fontSize: 28, fontWeight: FontWeight.bold)),
              const SizedBox(height: 32),
              TextField(
                controller: _passwordController,
                obscureText: _obscure,
                decoration: InputDecoration(
                  labelText: 'Master Password',
                  errorText: _error,
```

```dart
const _upper   = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
const _lower   = 'abcdefghijklmnopqrstuvwxyz';
const _digits  = '0123456789';
const _symbols = r'!@#$%^&*()-_=+[]{}|;:,.<>?';

/// Generates a random password according to the given constraints.
///
/// Throws [ArgumentError] if all character classes are disabled.
String generatePassword({
  required int length,
  bool uppercase = true,
  bool lowercase = true,
  bool digits = true,
  bool symbols = false,
}) {
  if (!uppercase && !lowercase && !digits && !symbols) {
    throw ArgumentError('At least one character class must be enabled');
  }
  final pool = [
    if (uppercase) _upper,
    if (lowercase) _lower,
    if (digits) _digits,
    if (symbols) _symbols,
  ].join();

  final rng = Random.secure();

  // Guarantee at least one character from each enabled class.
  final required = [
    if (uppercase) _upper[rng.nextInt(_upper.length)],
    if (lowercase) _lower[rng.nextInt(_lower.length)],
    if (digits) _digits[rng.nextInt(_digits.length)],
    if (symbols) _symbols[rng.nextInt(_symbols.length)],
  ];

  final remaining = List.generate(
    length - required.length,
    (_) => pool[rng.nextInt(pool.length)],
  );

  final chars = [...required, ...remaining]..shuffle(rng);
  return chars.join();
}
```

```dart
import 'package:onepass_app/models/category.dart';
import 'package:onepass_app/models/entry.dart';

/// Merges [remote] entries into [local] (keyed by id).
/// Rules:
/// 1. Delete-wins: if either side has deletedAt != null, mark as deleted.
/// 2. Last-write-wins: higher updatedAt wins for edit-edit conflicts.
Map<String, Entry> mergeEntries({
  required Map<String, Entry> local,
  required List<Entry> remote,
}) {
  final result = Map<String, Entry>.from(local);

  for (final remoteEntry in remote) {
    final localEntry = result[remoteEntry.id];

    if (localEntry == null) {
      result[remoteEntry.id] = remoteEntry;
      continue;
    }

    if (localEntry.deletedAt != null || remoteEntry.deletedAt != null) {
      final deletedAt = _earlierNonNull(localEntry.deletedAt, remoteEntry.deletedAt);
      final winner = localEntry.updatedAt.isAfter(remoteEntry.updatedAt)
          ? localEntry
          : remoteEntry;
      result[remoteEntry.id] = winner.copyWith(deletedAt: deletedAt);
      continue;
    }

    if (remoteEntry.updatedAt.isAfter(localEntry.updatedAt)) {
      result[remoteEntry.id] = remoteEntry;
    }
  }
  return result;
}

/// Merges [remote] categories into [local] using the same rules as entries.
Map<String, Category> mergeCategories({
  required Map<String, Category> local,
  required List<Category> remote,
}) {
  final result = Map<String, Category>.from(local);

  for (final remoteCat in remote) {
    final localCat = result[remoteCat.id];

    if (localCat == null) {
      result[remoteCat.id] = remoteCat;
      continue;
    }

    if (localCat.deletedAt != null || remoteCat.deletedAt != null) {
      final deletedAt = _earlierNonNull(localCat.deletedAt, remoteCat.deletedAt);
      final winner = localCat.updatedAt.isAfter(remoteCat.updatedAt) ? localCat : remoteCat;
      result[remoteCat.id] = winner.copyWith(deletedAt: deletedAt);
      continue;
    }

    if (remoteCat.updatedAt.isAfter(localCat.updatedAt)) {
      result[remoteCat.id] = remoteCat;
    }
  }
  return result;
}

DateTime? _earlierNonNull(DateTime? a, DateTime? b) {
  if (a == null) return b;
  if (b == null) return a;
  return a.isBefore(b) ? a : b;
}
```

```dart
import 'dart:io';
import 'dart:typed_data';
import 'package:flutter/services.dart';
import 'package:shelf/shelf.dart';
import 'package:shelf/shelf_io.dart' as shelf_io;

import 'package:onepass_app/crypto/crypto_service.dart';
import 'package:onepass_app/sync/sync_merge.dart';
import 'package:onepass_app/sync/sync_models.dart';
import 'package:onepass_app/vault/vault_service.dart';

class SyncServer {
  final VaultService _vault;
  late final String _expectedToken;
  HttpServer? _httpServer;

  SyncServer({required VaultService vault, required Uint8List keyBytes})
      : _vault = vault {
    _expectedToken = base64.encode(CryptoService.computeSyncToken(keyBytes));
  }

  /// The base64 Bearer token clients must present.
  String get authTokenBase64 => _expectedToken;

  /// Starts the HTTP server bound to the LAN IP and advertises via mDNS.
  Future<void> start({int port = 47200}) async {
    final lanIp = await _getLanIp();
    await _startHttp(InternetAddress(lanIp), port);
    await _advertiseMdns(port);
  }

  /// Starts the HTTP server on loopback only (for integration tests, no mDNS).
  Future<void> startForTesting({int port = 47200}) async {
    await _startHttp(InternetAddress.loopbackIPv4, port);
  }

  Future<void> _startHttp(InternetAddress address, int port) async {
    final router = Router()
      ..get('/sync',  _handleGetSync)
      ..post('/sync', _handlePostSync);

    final handler = Pipeline()
        .addMiddleware(_authMiddleware())
        .addHandler(router.call);

    _httpServer = await shelf_io.serve(handler, address, port);
  }

  Future<void> stop() async {
    await _stopMdns();
    await _httpServer?.close(force: true);
    _httpServer = null;
  }

  /// Returns the first non-loopback IPv4 address of this machine.
  /// Throws [StateError] if no LAN interface is found.
  Future<String> _getLanIp() async {
    for (final iface in await NetworkInterface.list()) {
      for (final addr in iface.addresses) {
        if (addr.type == InternetAddressType.IPv4 && !addr.isLoopback) {
          return addr.address;
        }
      }
    }
    throw StateError('No LAN interface found — connect to a WiFi network');
  }

  /// Advertises `_onepass._tcp` via macOS NetService through a MethodChannel.
  /// The native Swift implementation calls `NetService.publish()`.
  Future<void> _advertiseMdns(int port) async {
    const channel = MethodChannel('com.onepass/mdns');
    await channel.invokeMethod<void>('startAdvertising', {'port': port});
  }

  Future<void> _stopMdns() async {
    const channel = MethodChannel('com.onepass/mdns');
    try {
      await channel.invokeMethod<void>('stopAdvertising');
    } catch (_) {
      // Ignore errors on stop (server may not have been advertising).
    }
  }

  Middleware _authMiddleware() {
    return (Handler inner) => (Request request) async {
          final auth = request.headers['authorization'] ?? '';
          if (!auth.startsWith('Bearer ')) {
            return Response(401, body: jsonEncode({'error': 'missing token'}));
          }
          if (auth.substring('Bearer '.length) != _expectedToken) {
            return Response(401, body: jsonEncode({'error': 'invalid token'}));
          }
          return inner(request);
        };
  }

  Future<Response> _handleGetSync(Request request) async {
    final sinceMs = int.tryParse(request.url.queryParameters['since'] ?? '0') ?? 0;
    final since   = DateTime.fromMillisecondsSinceEpoch(sinceMs, isUtc: true);

    final entries    = await _vault.getEntriesSince(since);
    final categories = await _vault.getCategoriesSince(since);

    return Response.ok(
      jsonEncode(SyncResponse(
        entries: entries,
        categories: categories,
        serverTs: DateTime.now().toUtc().millisecondsSinceEpoch,
      ).toJson()),
      headers: {'content-type': 'application/json'},
    );
  }

  Future<Response> _handlePostSync(Request request) async {
    final payload = SyncPayload.fromJson(
      jsonDecode(await request.readAsString()) as Map<String, dynamic>,
    );

    final localEntries    = {for (final e in await _vault.getEntries()) e.id: e};
    final localCategories = {for (final c in await _vault.getCategories()) c.id: c};

    final merged = mergeEntries(local: localEntries, remote: payload.entries);
    final mergedCats = mergeCategories(local: localCategories, remote: payload.categories);

    for (final e in merged.values) await _vault.upsertEntry(e);
    for (final c in mergedCats.values) await _vault.upsertCategory(c);

    return Response.ok(
      jsonEncode({'ok': true}),
      headers: {'content-type': 'application/json'},
    );
  }
}
```

```dart
/// Performs bidirectional incremental sync with the Mac sync server.
class SyncClient {
  final VaultService _vault;
  late final String _token;

  SyncClient({required VaultService vault, required Uint8List keyBytes})
      : _vault = vault {
    _token = base64.encode(CryptoService.computeSyncToken(keyBytes));
  }

  /// Performs a full bidirectional sync.
  ///
  /// Returns the Mac's [serverTs] (Unix ms) to use as the [lastSyncTs] for
  /// the next sync call. Throws [SyncException] on any failure.
  Future<int> sync({required int lastSyncTs}) async {
    final macIp  = await _discoverMacIp();
    final base   = 'http://$macIp:$_syncPort';
    final headers = {'Authorization': 'Bearer $_token'};

    // ── Step 1: Pull changes from Mac ────────────────────────────────────
    final getRes = await http.get(
      Uri.parse('$base/sync?since=$lastSyncTs'),
      headers: headers,
    );
    _checkStatus(getRes, 'GET /sync');

    final syncResponse = SyncResponse.fromJson(
      jsonDecode(getRes.body) as Map<String, dynamic>,
    );

    final localEntries    = {for (final e in await _vault.getEntries()) e.id: e};
    final localCategories = {for (final c in await _vault.getCategories()) c.id: c};
    final mergedEntries    = mergeEntries(local: localEntries, remote: syncResponse.entries);
    final mergedCategories = mergeCategories(local: localCategories, remote: syncResponse.categories);
    for (final e in mergedEntries.values)    await _vault.upsertEntry(e);
    for (final c in mergedCategories.values) await _vault.upsertCategory(c);

    // ── Step 2: Push Android local changes to Mac ─────────────────────────
    // Use lastSyncTs (Mac server time) as the since threshold to find local changes.
    final since  = DateTime.fromMillisecondsSinceEpoch(lastSyncTs, isUtc: true);
    final postPayload = SyncPayload(
      entries:    await _vault.getEntriesSince(since),
      categories: await _vault.getCategoriesSince(since),
    );

    final postRes = await http.post(
      Uri.parse('$base/sync'),
      headers: {...headers, 'Content-Type': 'application/json'},
      body: jsonEncode(postPayload.toJson()),
    );
    _checkStatus(postRes, 'POST /sync');

    // Return serverTs so callers can persist it as the next lastSyncTs.
    return syncResponse.serverTs;
  }

  void _checkStatus(http.Response res, String label) {
    if (res.statusCode == 401) {
      throw SyncException('Sync auth failed — verify both devices use the same master password');
    }
    if (res.statusCode != 200) {
      throw SyncException('$label returned ${res.statusCode}');
    }
  }

  Future<String> _discoverMacIp() async {
    final client = MDnsClient();
    await client.start();
    try {
      await for (final PtrResourceRecord ptr in client
          .lookup<PtrResourceRecord>(ResourceRecordQuery.serverPointer(_mdnsServiceType))
          .timeout(_discoveryTimeout, onTimeout: (_) {
        throw SyncException('Mac not found on LAN — ensure OnePass is open on your Mac');
      })) {
        await for (final SrvResourceRecord srv in client
            .lookup<SrvResourceRecord>(ResourceRecordQuery.service(ptr.domainName))) {
          await for (final IPAddressResourceRecord ip in client
              .lookup<IPAddressResourceRecord>(ResourceRecordQuery.addressIPv4(srv.target))) {
            return ip.address.address;
          }
        }
      }
    } finally {
      client.stop();
    }
    throw SyncException('Mac not found on LAN');
  }
}

class SyncException implements Exception {
  final String message;
  const SyncException(this.message);

  @override
  String toString() => 'SyncException: $message';
}
```

```dart
class SyncNotifier extends AsyncNotifier<String> {

  Future<void> sync() async {

        SyncException('Vault is locked'),
        StackTrace.current,

      final client   = SyncClient(vault: vault, keyBytes: keyBytes);

      final serverTs = await client.sync(lastSyncTs: lastTs);

      // Persist serverTs (Mac clock) as the next sync baseline.
      ref.read(lastSyncTsProvider.notifier).state = serverTs;

      // Refresh UI providers.
      ref.invalidate(entriesProvider);
      ref.invalidate(filteredEntriesProvider);
      ref.invalidate(categoriesProvider);

      return 'Synced just now';
    });
  }
```

```dart
class ExportService {
  /// Encrypts [entries] and [categories] using AES-256-GCM with [keyBytes].
  /// Returns the raw bytes of the `.onepass` file (UTF-8 encoded JSON).
  static Future<Uint8List> export({
    required List<Entry> entries,
    required List<Category> categories,
    required Uint8List keyBytes,
  }) async {
    final rng = Random.secure();
    final iv = Uint8List.fromList(List.generate(12, (_) => rng.nextInt(256)));

    final payload = jsonEncode({
      'entries':    entries.map((e) => e.toJson()).toList(),
      'categories': categories.map((c) => c.toJson()).toList(),
    });
    final plaintext = Uint8List.fromList(utf8.encode(payload));

    final cipher = GCMBlockCipher(AESEngine());
    cipher.init(
      true, // encrypt
      AEADParameters(KeyParameter(keyBytes), 128, iv, Uint8List(0)),
    );
    final ciphertext = cipher.process(plaintext);

    final fileJson = jsonEncode({
      'version': 1,
      'iv':   base64.encode(iv),
      'data': base64.encode(ciphertext), // ciphertext includes the 16-byte GCM tag
    });
    return Uint8List.fromList(utf8.encode(fileJson));
  }
}
```

```dart
import 'dart:convert';
import 'dart:typed_data';
import 'package:pointycastle/export.dart';
import 'package:onepass_app/models/category.dart';
import 'package:onepass_app/models/entry.dart';

class ImportResult {
  final List<Entry> entries;
  final List<Category> categories;
  const ImportResult({required this.entries, required this.categories});
}

class ImportException implements Exception {
  final String message;
  const ImportException(this.message);
  @override
  String toString() => 'ImportException: $message';
}

class ImportService {
  /// Decrypts a `.onepass` file and returns its contents.
  /// Throws [ImportException] if decryption fails (wrong key or corrupt file).
  static Future<ImportResult> decrypt({
    required Uint8List fileBytes,
    required Uint8List keyBytes,
  }) async {
    try {
      final outer = jsonDecode(utf8.decode(fileBytes)) as Map<String, dynamic>;

      if ((outer['version'] as int? ?? 0) != 1) {
        throw ImportException('Unsupported file version');
      }

      final iv         = base64.decode(outer['iv'] as String);
      final ciphertext = base64.decode(outer['data'] as String);

      final cipher = GCMBlockCipher(AESEngine());
      cipher.init(
        false, // decrypt
        AEADParameters(KeyParameter(keyBytes), 128, iv, Uint8List(0)),
      );

      final plaintext = cipher.process(ciphertext);
      final payload = jsonDecode(utf8.decode(plaintext)) as Map<String, dynamic>;

      return ImportResult(
        entries: (payload['entries'] as List)
            .map((e) => Entry.fromJson(e as Map<String, dynamic>))
            .toList(),
        categories: (payload['categories'] as List)
            .map((c) => Category.fromJson(c as Map<String, dynamic>))
            .toList(),
      );
    } on ImportException {
      rethrow;
    } catch (e) {
      throw ImportException('Decryption failed — check the password is correct');
    }
  }
}
```

```dart
      await file.writeAsBytes(bytes);
      setState(() => _message = 'Exported to ${file.path}');
    } catch (e) {
      setState(() => _message = 'Export failed: $e');
    } finally {
      if (mounted) setState(() => _busy = false);
    }
  }
```

```kotlin
package com.onepass.onepass_app

import android.content.Context
import android.security.keystore.KeyGenParameterSpec
import android.security.keystore.KeyProperties
import android.util.Base64
import io.flutter.plugin.common.MethodCall
import io.flutter.plugin.common.MethodChannel
import java.security.KeyStore
import javax.crypto.Cipher
import javax.crypto.KeyGenerator
import javax.crypto.SecretKey
import javax.crypto.spec.GCMParameterSpec

class KeystorePlugin(private val context: Context) : MethodChannel.MethodCallHandler {
    private val KEYSTORE_ALIAS = "onepass_vault_key"
    private val PREFS_NAME     = "onepass_prefs"
    private val PREFS_KEY      = "onepass_wrapped_key"
    private val PREFS_IV_KEY   = "onepass_wrapped_iv"

    override fun onMethodCall(call: MethodCall, result: MethodChannel.Result) {
        when (call.method) {
            "storeKey" -> {
                val keyBase64 = call.argument<String>("key")
                    ?: return result.error("MISSING", "key missing", null)
                try {
                    storeKey(keyBase64)
                    result.success(null)
                } catch (e: Exception) {
                    result.error("KEYSTORE_ERROR", e.message, null)
                }
            }
            "loadKey" -> {
                try {
                    result.success(loadKey())
                } catch (e: Exception) {
                    result.error("KEYSTORE_ERROR", e.message, null)
                }
            }
            "hasKey" -> result.success(hasKey())
            else      -> result.notImplemented()
        }
    }

    private fun getOrCreateSecretKey(): SecretKey {
        val ks = KeyStore.getInstance("AndroidKeyStore").apply { load(null) }
        ks.getEntry(KEYSTORE_ALIAS, null)?.let {
            return (it as KeyStore.SecretKeyEntry).secretKey
        }
        val spec = KeyGenParameterSpec.Builder(
            KEYSTORE_ALIAS,
            KeyProperties.PURPOSE_ENCRYPT or KeyProperties.PURPOSE_DECRYPT
        ).setBlockModes(KeyProperties.BLOCK_MODE_GCM)
            .setEncryptionPaddings(KeyProperties.ENCRYPTION_PADDING_NONE)
            .setUserAuthenticationRequired(true)
            .setUserAuthenticationParameters(0, KeyProperties.AUTH_BIOMETRIC_STRONG)
            .build()
        return KeyGenerator.getInstance(KeyProperties.KEY_ALGORITHM_AES, "AndroidKeyStore")
            .apply { init(spec) }.generateKey()
    }

    private fun storeKey(keyBase64: String) {
        val secretKey = getOrCreateSecretKey()
        val cipher    = Cipher.getInstance("AES/GCM/NoPadding")
        cipher.init(Cipher.ENCRYPT_MODE, secretKey)
        val encrypted = cipher.doFinal(Base64.decode(keyBase64, Base64.DEFAULT))
        context.getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE).edit()
            .putString(PREFS_KEY,    Base64.encodeToString(encrypted, Base64.DEFAULT))
            .putString(PREFS_IV_KEY, Base64.encodeToString(cipher.iv,  Base64.DEFAULT))
            .apply()
    }

    private fun loadKey(): String? {
        val prefs = context.getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE)
        val encB64 = prefs.getString(PREFS_KEY,    null) ?: return null
        val ivB64  = prefs.getString(PREFS_IV_KEY, null) ?: return null
        val ks     = KeyStore.getInstance("AndroidKeyStore").apply { load(null) }
        val key    = (ks.getEntry(KEYSTORE_ALIAS, null) as? KeyStore.SecretKeyEntry)
            ?.secretKey ?: return null
        val cipher = Cipher.getInstance("AES/GCM/NoPadding")
        cipher.init(Cipher.DECRYPT_MODE, key,
            GCMParameterSpec(128, Base64.decode(ivB64, Base64.DEFAULT)))
        val plain = cipher.doFinal(Base64.decode(encB64, Base64.DEFAULT))
        return Base64.encodeToString(plain, Base64.DEFAULT)
    }

    private fun hasKey(): Boolean {
        return context.getSharedPreferences(PREFS_NAME, Context.MODE_PRIVATE)
            .contains(PREFS_KEY)
    }
}
```

```swift
import Foundation
import FlutterMacOS

class KeychainPlugin: NSObject, FlutterPlugin {
    static func register(with registrar: FlutterPluginRegistrar) {
        let channel = FlutterMethodChannel(
            name: "com.onepass/keystore",
            binaryMessenger: registrar.messenger
        )
        let instance = KeychainPlugin()
        registrar.addMethodCallDelegate(instance, channel: channel)
    }

    func handle(_ call: FlutterMethodCall, result: @escaping FlutterResult) {
        switch call.method {
        case "storeKey":
            guard let args = call.arguments as? [String: Any],
                  let keyBase64 = args["key"] as? String else {
                return result(FlutterError(code: "MISSING", message: "key missing", details: nil))
            }
            storeKey(keyBase64: keyBase64, result: result)
        case "loadKey":
            loadKey(result: result)
        case "hasKey":
            result(hasKey())
        default:
            result(FlutterMethodNotImplemented)
        }
    }

    private func storeKey(keyBase64: String, result: @escaping FlutterResult) {
        guard let data = Data(base64Encoded: keyBase64) else {
            return result(FlutterError(code: "INVALID", message: "invalid base64", details: nil))
        }
        let query: [String: Any] = [
            kSecClass as String:            kSecClassGenericPassword,
            kSecAttrAccount as String:      "onepass_vault_key",
            kSecAttrAccessible as String:   kSecAttrAccessibleWhenUnlockedThisDeviceOnly,
            kSecValueData as String:        data,
        ]
        SecItemDelete(query as CFDictionary)
        let status = SecItemAdd(query as CFDictionary, nil)
        if status == errSecSuccess { result(nil) }
        else { result(FlutterError(code: "KEYCHAIN", message: "SecItemAdd failed: \(status)", details: nil)) }
    }

    private func loadKey(result: @escaping FlutterResult) {
        let query: [String: Any] = [
            kSecClass as String:       kSecClassGenericPassword,
            kSecAttrAccount as String: "onepass_vault_key",
            kSecReturnData as String:  true,
            kSecMatchLimit as String:  kSecMatchLimitOne,
        ]
        var item: CFTypeRef?
        let status = SecItemCopyMatching(query as CFDictionary, &item)
        if status == errSecSuccess, let data = item as? Data {
            result(data.base64EncodedString())
        } else {
            result(nil)
        }
    }

    private func hasKey() -> Bool {
        let query: [String: Any] = [
            kSecClass as String:       kSecClassGenericPassword,
            kSecAttrAccount as String: "onepass_vault_key",
            kSecMatchLimit as String:  kSecMatchLimitOne,
        ]
        return SecItemCopyMatching(query as CFDictionary, nil) == errSecSuccess
    }
}
```

```swift
import Foundation

class MdnsPlugin: NSObject, FlutterPlugin, NetServiceDelegate {
    private var netService: NetService?

    static func register(with registrar: FlutterPluginRegistrar) {
        let channel = FlutterMethodChannel(
            name: "com.onepass/mdns",
            binaryMessenger: registrar.messenger
        )
        let instance = MdnsPlugin()
        registrar.addMethodCallDelegate(instance, channel: channel)
    }

    func handle(_ call: FlutterMethodCall, result: @escaping FlutterResult) {
        switch call.method {
        case "startAdvertising":
            guard let args = call.arguments as? [String: Any],
                  let port = args["port"] as? Int else {
                return result(FlutterError(code: "MISSING", message: "port missing", details: nil))
            }
            startAdvertising(port: port, result: result)
        case "stopAdvertising":
            netService?.stop()
            netService = nil
            result(nil)
        default:
            result(FlutterMethodNotImplemented)
        }
    }

    private func startAdvertising(port: Int, result: @escaping FlutterResult) {
        netService?.stop()
        netService = NetService(
            domain: "local.",
            type:   "_onepass._tcp.",
            name:   Host.current().localizedName ?? "OnePass",
            port:   Int32(port)
        )
        netService?.delegate = self
        netService?.publish()
        result(nil)
    }

    func netServiceDidPublish(_ sender: NetService) {}

    func netService(_ sender: NetService, didNotPublish errorDict: [String: NSNumber]) {
        // Log failure silently — sync is best-effort.
        print("mDNS publish failed: \(errorDict)")
    }
}
```
